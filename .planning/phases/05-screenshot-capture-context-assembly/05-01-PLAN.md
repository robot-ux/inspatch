  const payload: Record<string, unknown> = {

    className: typeof el.className === 'string' ? el.className : '',

    boundingRect: {
      x: Math.round(rect.x),

      width: Math.round(rect.width),
      height: Math.round(rect.height),
    },
  };

}

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | computedStyles payload | mitigate | Zod schema validates computedStyles as `z.record(z.string(), z.string())` — only string key/values allowed. Property names are from a fixed curated list in content script, not user-supplied. |
| T-05-02 | Denial of Service | computedStyles size | mitigate | Curated list is 30 properties max with empty/default filtering. Payload cannot grow unbounded. |
| T-05-03 | Tampering | devicePixelRatio | accept | Value comes from `window.devicePixelRatio` (browser API). Page cannot override this. If spoofed, worst case is mis-cropped screenshot — no security impact. |
</threat_model>
