**1. Create `packages/server/src/queue.ts` — Serial request queue (per D-14, D-15, D-16):**

```typescript
import type { ServerWebSocket } from "bun";
import type { ChangeRequest } from "@inspatch/shared";

interface QueuedRequest {
  request: ChangeRequest;
  ws: ServerWebSocket<WSData>;
  enqueuedAt: number;
}

export type WSData = {
  connectedAt: number;
  id: string;
};

export class RequestQueue {
  private queue: QueuedRequest[] = [];
  private processing = false;

  get length(): number { return this.queue.length; }
  get isProcessing(): boolean { return this.processing; }

  enqueue(request: ChangeRequest, ws: ServerWebSocket<WSData>): void {
    this.queue.push({ request, ws, enqueuedAt: Date.now() });
    this.process();
  }

  private async process(): Promise<void> {
    if (this.processing || this.queue.length === 0) return;
    this.processing = true;

    const item = this.queue.shift()!;
    const { request, ws } = item;

    // Send "queued" acknowledgment (D-16: placeholder for Phase 6)
    this.sendStatus(ws, request.requestId, "queued", "Request received and queued for processing");

    // Phase 6 wires actual Claude Code invocation here
    // For now: immediately complete with placeholder
    this.sendStatus(ws, request.requestId, "complete", "Processing pipeline not yet connected");

    this.processing = false;
    this.process();
  }

  private sendStatus(ws: ServerWebSocket<WSData>, requestId: string | undefined, status: string, message: string): void {
    const payload: Record<string, unknown> = { type: "status_update", status, message };
    if (requestId) payload.requestId = requestId;
    ws.send(JSON.stringify(payload));
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Extension → Server | Side panel sends WebSocket messages to local server; untrusted input crosses here |
| Server → Extension | Server sends responses back; could be spoofed if another process binds port |
| Local process → Server | Any process on localhost can connect to 127.0.0.1:9377 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | server.ts WebSocket | accept | 127.0.0.1 binding (D-09) limits to local processes only. Token auth deferred — low risk for local dev tool. |
| T-03-02 | Tampering | server.ts message handler | mitigate | Zod `safeParse()` validates ALL incoming messages (D-11). Invalid JSON caught in try/catch. Invalid schema rejected with error response. |
| T-03-03 | Denial of Service | server.ts | mitigate | Serial queue (D-14) naturally throttles processing. `idleTimeout: 60` drops stale connections. Message must pass Zod validation before reaching queue. |
| T-03-04 | Tampering | queue.ts | mitigate | Queue only accepts messages that passed Zod ChangeRequest validation. No raw user input reaches processing logic. |
</threat_model>
